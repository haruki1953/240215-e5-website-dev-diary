    await UsersE5SharedInfo.create({

    }, { transaction });

    await transaction.commit();

    // 返回新创建的用户
    return user;
  } catch (error) {
    // 回滚事务

## 使用 sharp 压缩图片
安装 sharp ：pnpm add sharp@0.33.2
```js
const sharp = require('sharp');
const fs = require('fs').promises;
const path = require('path');
const { avatarConfig } = require('./config');

// 处理头像的函数，只需传入文件名
async function processAvatar(filename) {
  // 输入文件
  const inputFilePath = path.join(avatarConfig.uploadPath, filename);
  // 保存文件 拼接.jpg
  const saveFileName = `${filename}.jpg`;
  const saveFilePath = path.join(avatarConfig.savePath, saveFileName);
  try {
    // 读取输入图片文件
    const inputBuffer = await fs.readFile(inputFilePath);

    // 使用 sharp 库处理图片
    const outputBuffer = await sharp(inputBuffer)
      .resize(avatarConfig.size, avatarConfig.size, { fit: 'cover' }) // 调整为 256x256 的正方形
      .jpeg({ quality: avatarConfig.quality }) // 设置 JPEG 图片质量
      .toBuffer(); // 转换为 Buffer

    // 将处理后的图片数据写入到输出文件中
    await fs.writeFile(saveFilePath, outputBuffer);
    // 删除原始图片
    fs.unlink(inputFilePath);
    // 返回保存的文件名
    return saveFileName;
  } catch (error) {
    fs.unlink(inputFilePath); // 删除输入文件
    fs.unlink(saveFilePath); // 删除输出文件
    throw new ClientError('图片处理失败');
  }
}
```

在config文件导出 头像配置 avatarConfig
```js
// 头像配置
exports.avatarConfig = {
  uploadPath: path.join(__dirname, './uploads'), // 上传时的临时保存路径
  savePath: path.join(__dirname, './uploads/avatar'), // 处理后的保存路径
  size: 256, // 图片大小
  quality: 64, // 图片质量
};
```

## 备份文件命名
如：database.StartupBackup.20230406152130876.sqlite
1. 倒数第三个后缀为备份类型
2. 倒数第二个后缀为`toISOString`返回的字符串剔除非数字（UTC）

**备份类型有三种**
1. StartupBackup 启动时备份，保留5份
2. DailyBackup 每天备份，保留5份
3. MonthlyBackup 每月备份


## `node-cron` Node.js中执行定时任务
安装：pnpm add node-cron
导入：const cron = require('node-cron');
```js
// 启动备份系统（在app.js中调用）
const startBackupSystem = async () => {
  // 启动时备份
  await startupBackupDataBase();

  // node-cron 定时执行备份任务
  // 每日备份任务
  cron.schedule(dbConfig.backupCron.DailyBackup, async () => {
    await dailyBackupDataBase();
  });
  // 每月备份任务
  cron.schedule(dbConfig.backupCron.MonthlyBackup, async () => {
    await monthlyBackupDataBase();
  });
};

// dbConfig:
// backupCron: { // 备份频率Cron表达式，注意：UTC时间
//   DailyBackup: '0 1 * * *', // 每天凌晨1点执行
//   MonthlyBackup: '1 1 1 * *', // 每月1号凌晨1点1分执行
// },
```

## 有一个奇怪的现象
```js
// 查询所有用户信息，排除密码和备注字段，排除last_login,updatedAt字段
users = await User.findAll({
  attributes: { exclude: ['password_hash', 'note', 'last_login', 'updatedAt'] },
  raw: true, // 指示返回原始的 JSON 对象
});
```
raw: true 会导致火狐浏览器上 **E5订阅日期** 显示不正常（会不显示，进度条是已过期），而谷歌浏览器则没问题。去掉raw: true则全部正常
> ？可能的原因是**E5订阅日期**字段在数据库中的数据类型和在浏览器中需要的数据类型不一致。例如，如果**E5订阅日期**在数据库中是一个时间戳，但在浏览器中需要的是一个日期字符串，那么在使用`raw: true`选项时，Sequelize将不会自动将时间戳转换为日期字符串，导致在火狐浏览器上显示不正常。

破案了，确实时间格式会有区别，可能火狐的new Date不支持这种格式的
```
正常：
e5_expiration_date	"2024-06-07T16:00:00.000Z"

raw: true：
e5_expiration_date	"2024-06-07 16:00:00.000 +00:00"
```
